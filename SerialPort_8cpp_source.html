818  inline
819  void
820  SerialPort::Implementation::Open(const std::string& fileName,
821  const std::ios_base::openmode& openMode)
822  {
823  // Throw an exception if the port is already open.
824  if (this->IsOpen())
825  {
826  throw AlreadyOpen(ERR_MSG_PORT_ALREADY_OPEN) ;
827  }
828 
829  // We only allow three different combinations of ios_base::openmode so
830  // we can use a switch here to construct the flags to be used with the
831  // open() system call. Since we are dealing with the serial port we
832  // need to use the O_NOCTTY option.
833  int flags = (O_NOCTTY | O_NONBLOCK) ; // NOLINT (hicpp-signed-bitwise)
834 
835  if (openMode == (std::ios_base::in | std::ios_base::out))
836  {
837  flags |= O_RDWR ;
838  }
839  else if (openMode == std::ios_base::in)
840  {
841  flags |= O_RDONLY ;
842  }
843  else if (openMode == std::ios_base::out)
844  {
845  flags |= O_WRONLY ;
846  }
847  else
848  {
849  throw OpenFailed {"Invalid or unsupported open mode"} ;
850  }
851 
852  // Try to open the serial port.
853  // NOLINTNEXTLINE (cppcoreguidelines-pro-type-vararg)
854  mFileDescriptor = call_with_retry(open, fileName.c_str(), flags) ;
855 
856  if (this->mFileDescriptor < 0)
857  {
858  close(this->mFileDescriptor) ;
859  throw OpenFailed(std::strerror(errno)) ;
860  }
861 
862  // Set the serial port to exclusive access to this process.
863  // NOLINTNEXTLINE (cppcoreguidelines-pro-type-vararg)
864  if (call_with_retry(ioctl,
865  this->mFileDescriptor,
866  TIOCEXCL) == -1)
867  {
868  throw std::runtime_error(std::strerror(errno)) ;
869  }
870 
871  // Save the current settings of the serial port so they can be
872  // restored when the serial port is closed.
873  if (tcgetattr(this->mFileDescriptor,
874  &mOldPortSettings) < 0)
875  {
876  throw OpenFailed(std::strerror(errno)) ;
877  }
878 
879  // Set up the default configuration for the serial port.
880  this->SetDefaultSerialPortParameters() ;
881 
882  // Flush the input and output buffers associated with the port.
883  this->FlushIOBuffers() ;
884  }

889  {

892  {
893  throw NotOpen(ERR_MSG_PORT_NOT_OPEN) ;
894  }
895

901  throw std::runtime_error(std::strerror(errno)) ;

909  }

912  mFileDescriptor = -1 ;

1042  inline
1043  void
1044  SerialPort::Implementation::SetBaudRate(const BaudRate& baudRate)
1045  {
1046  // Throw an exception if the serial port is not open.
1047  if (not this->IsOpen())
1048  {
1049  throw NotOpen(ERR_MSG_PORT_NOT_OPEN) ;
1050  }
1051 
1052  // Get the current serial port settings.
1053  termios port_settings {} ;
1054  std::memset(&port_settings, 0, sizeof(port_settings)) ;
1055 
1056  if (tcgetattr(this->mFileDescriptor,
1057  &port_settings) < 0)
1058  {
1059  throw std::runtime_error(std::strerror(errno)) ;
1060  }
1061 
1062  // Set the baud rate for both input and output.
1063  if (0 != cfsetspeed(&port_settings, static_cast<speed_t>(baudRate)))
1064  {
1065  // If applying the baud rate settings fail, throw an exception.
1066  throw std::runtime_error(ERR_MSG_INVALID_BAUD_RATE) ;
1067  }
1068 
1069  // Apply the modified settings.
1070  if (tcsetattr(this->mFileDescriptor,
1071  TCSANOW,
1072  &port_settings) < 0)
1073  {
1074  // If applying the settings fails, throw an exception.
1075  throw std::runtime_error(std::strerror(errno)) ;
1076  }
1077 
1078  // Set the time interval value (us) required for one byte of data to arrive.
1079  mByteArrivalTimeDelta = (BITS_PER_BYTE * MICROSECONDS_PER_SEC) / GetBitRate(baudRate) ;
1080  }

1082  inline
1083  BaudRate
1084  SerialPort::Implementation::GetBaudRate() const
1085  {
1086  // Throw an exception if the serial port is not open.
1087  if (not this->IsOpen())
1088  {
1089  throw NotOpen(ERR_MSG_PORT_NOT_OPEN) ;
1090  }
1091 
1092  // Get the current serial port settings.
1093  termios port_settings {} ;
1094  std::memset(&port_settings, 0, sizeof(port_settings)) ;
1095 
1096  if (tcgetattr(this->mFileDescriptor,
1097  &port_settings) < 0)
1098  {
1099  throw std::runtime_error(std::strerror(errno)) ;
1100  }
1101 
1102  // Read the input and output baud rates.
1103  const auto input_baud = cfgetispeed(&port_settings) ;
1104  const auto output_baud = cfgetospeed(&port_settings) ;
1105 
1106  // Make sure that the input and output baud rates are
1107  // equal. Otherwise, we do not know which one to return.
1108  if (input_baud != output_baud)
1109  {
1110  throw std::runtime_error(ERR_MSG_INVALID_BAUD_RATE) ;
1111  return BaudRate::BAUD_INVALID ;
1112  }
1113 
1114  // Obtain the input baud rate from the current settings.
1115  return BaudRate(input_baud) ;
1116  }

1118  inline
1119  int
1120  SerialPort::Implementation::GetBitRate(const BaudRate& baudRate) const
1121  {
1122  int baud_rate_as_int = 1 ;
1123 
1124  switch (baudRate)
1125  {
