920  inline
921  std::streamsize
922  SerialStreamBuf::Implementation::xsputn(const char_type* character,
923  std::streamsize numberOfBytes)
924  {
925  // Throw an exception if the serial port is not open.
926  if (!this->IsOpen())
927  {
928  throw NotOpen(ERR_MSG_PORT_NOT_OPEN) ;
929  }
930 
931  // If n is less than 1 there is nothing to accomplish.
932  if (numberOfBytes <= 0)
933  {
934  return 0 ;
935  }
936 
937  // Write the n characters to the serial port.
938  //
939  // :TODO: Consider using mSerialPort.Write() here instead of writing
940  // to the file descriptor directly.
941  const auto fd = mSerialPort.GetFileDescriptor() ;
942  ssize_t result = call_with_retry(write, fd, character, numberOfBytes) ;
943 
944  // If the write failed then return 0.
945  if (result <= 0)
946  {
947  return 0 ;
948  }
949 
950  // Otherwise, return the number of bytes actually written.
951  return result;
952  }

954  inline
955  std::streamsize
956  SerialStreamBuf::Implementation::xsgetn(char_type* character,
957  std::streamsize numberOfBytes)
958  {
959  // Throw an exception if the serial port is not open.
960  if (!this->IsOpen())
961  {
962  throw NotOpen(ERR_MSG_PORT_NOT_OPEN) ;
963  }
964 
965  // If n is less than 1 there is nothing to accomplish.
966  if (numberOfBytes <= 0)
967  {
968  return 0 ;
969  }
970 
971  if (character == nullptr) {
972  return 0 ;
973  }
974 
975  // Try to read up to n characters in the array s.
976  ssize_t result = -1;
977 
978  // If a putback character is available, then we need to read only
979  // n-1 character.
980  if (mPutbackAvailable)
981  {
982  // Put the mPutbackChar at the beginning of the array 's'.
983  // Increment result to indicate that a character has been placed in s.
984  character[0] = mPutbackChar;
985  result++;
986 
987  // The putback character is no longer available.
988  mPutbackAvailable = false;
989 
990  // If we need to read more than one character, then call read()
991  // and try to read numberOfBytes-1 more characters and put them
992  // at location starting from &character[1].
993  if (numberOfBytes > 1)
994  {
995  //
996  // :TODO: Consider using mSerialPort.Read() here instead of
997  // using the file descriptor.
998  //
999  const auto fd = mSerialPort.GetFileDescriptor() ;
1000  result = call_with_retry(read, fd, &character[1], numberOfBytes-1) ;
1001 
1002  // If read was successful, then we need to increment result by
1003  // one to indicate that the putback character was prepended to
1004  // the array, s. If read failed then leave result at -1.
1005  if (result != -1)
1006  {
1007  result++;
1008  }
1009  }
1010  }
1011  else
1012  {
1013  // If no putback character is available then we try to
1014  // read numberOfBytes characters.
1015  //
1016  // :TODO: Consider using mSerialPort.Read() here instead of
1017  // using the file descriptor.
1018  const auto fd = mSerialPort.GetFileDescriptor() ;
1019  result = call_with_retry(read, fd, character, numberOfBytes) ;
1020  }
1021 
1022  // If result == -1 then the read call had an error, otherwise, if
1023  // result == 0 then we could not read the characters. In either
1024  // case, we return 0 to indicate that no characters could be read
1025  // from the serial port.
1026  if (result <= 0)
1027  {
1028  return 0 ;
1029  }
1030 
1031  // Return the number of characters actually read from the serial port.
1032  return result;
1033  }

1035  inline
1036  std::streambuf::int_type
1037  SerialStreamBuf::Implementation::overflow(const int_type character)
1038  {
1039  // Throw an exception if the serial port is not open.
1040  if (!this->IsOpen())
1041  {
1042  throw NotOpen(ERR_MSG_PORT_NOT_OPEN) ;
1043  }
1044 
1045  // Try to write the specified character to the serial port.
1046  if (traits_type::eq_int_type(character, traits_type::eof()))
1047  {
1048  // If character is the eof character then we do nothing.
1049  return traits_type::eof() ;
1050  }
1051 
1052  // Otherwise we write the character to the serial port.
1053  //
1054  // :TODO: Consider using a method of SerialPort class here instead
1055  // of using the file descriptor.
1056  const auto fd = mSerialPort.GetFileDescriptor() ;
1057  char out_char = traits_type::to_char_type(character) ;
1058  ssize_t result = call_with_retry(write, fd, &out_char, 1) ;
1059 
1060  // If the write failed then return eof.
1061  if (result <= 0)
1062  {
1063  return traits_type::eof() ;
1064  }
1065 
1066  // Otherwise, return something other than eof().
1067  return traits_type::not_eof(character) ;
1068  }

1070  inline
1071  std::streambuf::int_type
1072  SerialStreamBuf::Implementation::underflow()
1073  {
1074  // Throw an exception if the serial port is not open.
1075  if (!this->IsOpen())
1076  {
1077  throw NotOpen(ERR_MSG_PORT_NOT_OPEN) ;
1078  }
1079 
1080  // Read the next character from the serial port.
1081  char next_char = 0 ;
1082  ssize_t result = -1;
1083 
1084  // If a putback character is available then we return that
1085  // character. However, we are not supposed to change the value of
1086  // gptr() in this routine so we leave mPutbackAvailable set to true.
1087  if (mPutbackAvailable)
1088  {
1089  next_char = mPutbackChar;
1090  }
1091  else
1092  {
1093  // If no putback character is available then we need to read one
1094  // character from the serial port.
1095  //
1096  // :TODO: Consider using a method of SerialPort class here instead
1097  // of using the file descriptor.
1098  //
1099  const auto fd = mSerialPort.GetFileDescriptor() ;
1100  result = call_with_retry(read, fd, &next_char, 1) ;
1101
